\pgfdeclarepattern{
  name=hatch,
  parameters={\hatchsize,\hatchangle,\hatchlinewidth},
  bottom left={\pgfpoint{0pt}{0pt}},
  top right={\pgfpoint{\hatchsize}{\hatchsize}},
  tile size={\pgfpoint{\hatchsize}{\hatchsize}},
  tile transformation={\pgftransformrotate{\hatchangle}},
  code={
    \pgfsetlinewidth{\hatchlinewidth}
    \pgfpathmoveto{\pgfpoint{0pt}{0pt}}
    \pgfpathlineto{\pgfpoint{\hatchsize}{\hatchsize}}
    \pgfusepath{stroke}
    \pgfpathmoveto{\pgfpoint{\hatchsize-1pt}{-1pt}}
    \pgfpathlineto{\pgfpoint{\hatchsize+1pt}{1pt}}
    \pgfusepath{stroke}
    \pgfpathmoveto{\pgfpoint{-1pt}{\hatchsize-1pt}}
    \pgfpathlineto{\pgfpoint{1pt}{\hatchsize+1pt}}
    \pgfusepath{stroke}
  }
}

\tikzset{
  hatch size/.store in=\hatchsize,
  hatch angle/.store in=\hatchangle,
  hatch line width/.store in=\hatchlinewidth,
  hatch size=5pt,
  hatch angle=0pt,
  hatch line width=.5pt,
}
