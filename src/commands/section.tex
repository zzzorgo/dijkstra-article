\newcommand{\sectionabstract}[2]{
    {\setstretch{1.7}
        \twocolumn[
            \begin{@twocolumnfalse}
                \section{#1}
                \Large{\myfont{\color{accentColor}{\vspace{0.7cm}#2\vspace{1.8cm}}}}
            \end{@twocolumnfalse}
        ]
    }
}

\newcommand{\summarysectiontitle}[1]{
    {\setstretch{1.7}
        \twocolumn[
            \begin{@twocolumnfalse}
                \section*{#1}
            \end{@twocolumnfalse}
        ]
    }
}
